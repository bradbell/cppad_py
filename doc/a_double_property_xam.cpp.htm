# include <cstdio>
# include <cppad/py/cppad_py.hpp>

bool a_double_property_xam(void) {
    using cppad_py::a_double;
    //
    // initialize return variable
    bool ok = true;
    //------------------------------------------------------------------------
    a_double a3(3.0);
    //
    ok = ok && a3   == 3.0;
    ok = ok && a3.parameter();
    ok = ok && ! a3.variable();
    //
    // near_equal
    a_double r3 = a3.sqrt() ;
    ok = ok && a3.near_equal( r3 * r3) ;;
    //
    // var2par
    a_double p3 = a3.var2par();
    ok &= p3.value() == 3.0;
    //
    return( ok );
}
