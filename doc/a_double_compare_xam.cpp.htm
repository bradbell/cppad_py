# include <cstdio>
# include <cppad/py/cppad_py.hpp>

bool a_double_compare_xam(void) {
    using cppad_py::a_double;
    bool ok = true;
    a_double a2(2.0);
    a_double a3(3.0);
    //------------------------------------------------------------------------
    ok = ok && a2   <  a3;
    ok = ok && a2   <= a3;
    ok = ok && a3 >  a2;
    ok = ok && a3 >= a2;
    ok = ok && a3 != a2;
    ok = ok && a3 == a3;
    //
    ok = ok && ! (a2 >  a3) ;
    ok = ok && ! (a2 >= a3) ;
    ok = ok && ! (a2 == a3) ;
    //------------------------------------------------------------------------
    ok = ok && a2   <  3.0;
    ok = ok && a2   <= 3.0;
    ok = ok && a3 >  2.0;
    ok = ok && a3 >= 2.0;
    ok = ok && a3 != 2.0;
    ok = ok && a3 == 3.0;
    //
    ok = ok && ! (a2 >  3.0) ;
    ok = ok && ! (a2 >= 3.0) ;
    ok = ok && ! (a2 == 3.0) ;
    //
    return( ok );
}
